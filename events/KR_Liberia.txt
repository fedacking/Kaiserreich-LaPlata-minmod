			factor = 30
		}
		name = lib.6.b

		set_politics = {ruling_party = paternal_autocrat}
		add_political_power = -100
		end_puppet = LBR
		hidden_effect = {
			if = { limit = {
			LBR = {is_subject_of = GER}
			}
			GER = {country_event = {id = lib.11 days = 7} }
			}
			if = { limit = {
			LBR = {is_subject_of = USA}
			}
			USA = {country_event = {id = lib.11 days = 7} }
			}
			if = { limit = {
			LBR = {is_subject_of = NFA}
