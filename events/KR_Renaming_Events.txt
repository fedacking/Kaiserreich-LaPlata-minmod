###########################
# Renaming Events
###########################
#
# Each event id should be 4 digits long
# Add extra digits where necessary - state 12 is 1200 - 1209 in event id's
# Use the first event for a state (for example, '1200') for the original owner
# Then subsequent id's are for other countries that can own & rename that state
#
###########################

add_namespace = rename

### 9 - Cechy
# Restore Czech Name
country_event = {
	id = rename.900
	hidden = yes

	trigger = {
		OR = {
			tag = BOM
			tag = CZE
		}
		owns_state = 9
		NOT = { has_country_flag = renamed_state_9 }
		# This state has been renamed by another country
		has_global_flag = renamed_state_9
	}

	mean_time_to_happen = {
		days = 3
	}

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_state_9 }
				clr_country_flag = renamed_state_9
			}
		}
	}

	option = {
		name = rename.900.a
		9 = { reset_state_name = yes }
		reset_province_name = 11542
		reset_province_name = 6418

		set_country_flag = renamed_state_9
		clr_global_flag = renamed_state_9
	}
}

	hidden = yes

	trigger = {
		tag = SER
		765 = { is_controlled_by = ROOT }
		NOT = { has_country_flag = renamed_kosova }
	}

	mean_time_to_happen = { days = 3 }

	immediate = {
		hidden_effect = {
			every_country = {
				limit = { has_country_flag = renamed_kosova }
				clr_country_flag = renamed_kosova
			}
		}
	}

	option = {
		name = rename.76501.a
		765 = { set_state_name = "Kosovo" }

		set_country_flag = renamed_kosova
		set_global_flag = renamed_kosova
	}
}
