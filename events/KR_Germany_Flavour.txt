news_event = {
	id = gerfla.3
	title = gerfla.3.t
	desc = gerfla.3.d
	picture = GER_hoffmann_richthofen

	trigger = {
		tag = GER
		OR = {
			controls_state = 546
			any_country = {
				is_subject_of = GER
				controls_state = 546
			}
		}
		NOT = {
			has_country_flag = Manfred_Freiherr_von_Richthofen_dead
		}
		date > 1939.6.9
	}

	mean_time_to_happen = { days = 2 }

	fire_only_once = yes

	option = {
		name = gerfla.3.a
		hidden_effect = {
			random_list = {
				50 = { GER = { news_event = { id = gerfla.4 days = 4 } } }
				50 = { GER = { news_event = { id = gerfla.5 days = 4 } } }
			}
		}
	}
}
